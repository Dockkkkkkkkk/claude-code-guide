## 内容与知识管理

```bash
# Notion
claude mcp add notion \
  -e OPENAPI_MCP_HEADERS='{"Authorization":"Bearer secret_xxx","Notion-Version":"2022-06-28"}' \
  -- npx -y @notionhq/notion-mcp-server

# 文件系统增强
claude mcp add filesystem \
  -- npx -y @modelcontextprotocol/server-filesystem ~/Documents

# Memory 持久化记忆
claude mcp add memory \
  -- npx -y @modelcontextprotocol/server-memory

# Obsidian 知识库
claude mcp add obsidian \
  -e OBSIDIAN_VAULT_PATH="/path/to/vault" \
  -- npx -y @obsidian-mcp/server
```

## 数据库

```bash
# SQLite
claude mcp add sqlite \
  -- npx -y @modelcontextprotocol/server-sqlite ~/data/db.sqlite

# PostgreSQL
claude mcp add postgres \
  -- npx -y @modelcontextprotocol/server-postgres \
  "postgresql://user:password@localhost/dbname"

# MySQL
claude mcp add mysql \
  -e MYSQL_HOST=localhost \
  -e MYSQL_USER=root \
  -e MYSQL_PASSWORD=password \
  -e MYSQL_DATABASE=mydb \
  -- npx -y @smithery/mcp-server-mysql

# Redis
claude mcp add redis \
  -e REDIS_URL=redis://localhost:6379 \
  -- npx -y @smithery/mcp-server-redis
```

```bash

claude mcp add github \
  -e GITHUB_PERSONAL_ACCESS_TOKEN=ghp_xxxx \
  -- npx -y @modelcontextprotocol/server-github

  -e GITLAB_PERSONAL_ACCESS_TOKEN=你的Token \
```
